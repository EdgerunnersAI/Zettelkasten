  ```sql
  -- Workspace-scope projection: feeds /api/graph viz (broad signal, all event types)
  CREATE MATERIALIZED VIEW IF NOT EXISTS kg.kg_edge_viz_weights AS
    SELECT workspace_id, source_node_id, target_node_id,
           COUNT(*)            AS event_count,
           SUM(weight_delta)   AS viz_weight,
           MAX(created_at)     AS last_touched_at
      FROM rag.retrieval_feedback_events
     WHERE event_type IN ('impression','click','dwell','cite','expand','follow_up')
       AND source_node_id IS NOT NULL AND target_node_id IS NOT NULL
     GROUP BY 1,2,3;
  CREATE UNIQUE INDEX IF NOT EXISTS kgevw_pk
    ON kg.kg_edge_viz_weights (workspace_id, source_node_id, target_node_id);

  -- Kasten-scope projection: feeds RAG ranker boost (retrieval-relevant events only)
  CREATE MATERIALIZED VIEW IF NOT EXISTS rag.kasten_retrieval_edge_signals AS
    SELECT workspace_id, kasten_id, source_node_id, target_node_id,
           SUM(weight_delta) FILTER (WHERE event_type IN ('cite','accept'))   AS positive_signal,
           SUM(weight_delta) FILTER (WHERE event_type = 'reject')             AS negative_signal,
           COUNT(*)                                                            AS event_count,
           MAX(created_at)                                                     AS last_event_at
      FROM rag.retrieval_feedback_events
     WHERE kasten_id IS NOT NULL
       AND source_node_id IS NOT NULL AND target_node_id IS NOT NULL
     GROUP BY 1,2,3,4;
  CREATE UNIQUE INDEX IF NOT EXISTS kres_pk
    ON rag.kasten_retrieval_edge_signals (workspace_id, kasten_id, source_node_id, target_node_id);

  -- Note: Postgres MVs do NOT inherit RLS (PG ≤17). MV reads are filtered at query time
  -- by joining workspace_members in the calling repository code, NOT by RLS on the MV.
  ```

- [Crunchy Data — Indexing JSONB in Postgres](https://www.crunchydata.com/blog/indexing-jsonb-in-postgres) — `jsonb_path_ops` ~3× smaller than default `jsonb_ops` for `@>` containment.
- [pganalyze — Postgres GIN Indexes](https://pganalyze.com/blog/gin-index) — partial GIN per kind, not blanket.
- [Notion — The data model behind Notion's flexibility](https://www.notion.com/blog/data-model-behind-notion) — `kind` discriminator + payload, schema_ver per row.
- [Neo4j 2025 dynamic labels and types](https://github.com/neo4j/neo4j/wiki/Neo4j-2025-changelog) — `entity_type` as text not enum; runtime-resolved.

  ```sql
  -- supabase/website/_v2/33_introspect_auth_users_dependents.sql
  CREATE OR REPLACE FUNCTION core.introspect_auth_users_dependents()
  RETURNS TABLE(schema_name text, table_name text, fk_column text)
  LANGUAGE sql STABLE SECURITY DEFINER SET search_path = public AS $$
    WITH RECURSIVE deps AS (
      SELECT kcu.table_schema::text AS schema_name,
             kcu.table_name::text   AS table_name,
             kcu.column_name::text  AS fk_column,
             ccu.table_schema::text AS ref_schema,
             ccu.table_name::text   AS ref_table
        FROM information_schema.key_column_usage kcu
        JOIN information_schema.referential_constraints rc
          ON kcu.constraint_name = rc.constraint_name
         AND kcu.constraint_schema = rc.constraint_schema
        JOIN information_schema.constraint_column_usage ccu
          ON ccu.constraint_name = rc.unique_constraint_name
         AND ccu.constraint_schema = rc.unique_constraint_schema
       WHERE ccu.table_schema = 'auth' AND ccu.table_name = 'users'
      UNION
      SELECT kcu.table_schema::text, kcu.table_name::text, kcu.column_name::text,
             ccu.table_schema::text, ccu.table_name::text
        FROM information_schema.key_column_usage kcu

        JOIN information_schema.constraint_column_usage ccu

        JOIN deps d ON d.schema_name = ccu.table_schema AND d.table_name = ccu.table_name

  $$;
  ```
